     4→// Convert english digits to Persian
     5→export const fa = (n: string | number): string => {
     6→  const faDigits = ["۰", "۱", "۲", "۳", "۴", "۵", "۶", "۷", "۸", "۹"];
     7→  return String(n).replace(/\d/g, (d) => faDigits[parseInt(d)]);
     8→};
     9→
    10→export const formatToman = (n: number): string => {
    11→  return fa(n.toLocaleString("en-US"));
    12→};
